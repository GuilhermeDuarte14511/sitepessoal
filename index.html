public class VertemPaymentsService
{
    private readonly IMessageBus bus;
    private readonly ISettlementGateway gateway;

    public VertemPaymentsService(IMessageBus bus, ISettlementGateway gateway)
    {
        this.bus = bus;
        this.gateway = gateway;
    }

    public async Task<Result> ExecutePixAsync(PixRequest request)
    {
        await bus.PublishAsync(new PaymentInitiated(request));
        var approval = await gateway.ValidateLimitAsync(request);
        return approval.IsApproved
            ? Result.Success()
            : Result.Fail("Límite insuficiente");
    }
}
